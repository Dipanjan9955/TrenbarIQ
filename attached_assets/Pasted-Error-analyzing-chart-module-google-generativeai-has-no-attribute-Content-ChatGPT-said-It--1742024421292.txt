import google.generativeai as genai
import base64
from flask import Flask, request, jsonify
from flask_cors import CORS

# Set up Gemini API Key
genai.configure(api_key="YOUR_GEMINI_API_KEY")

app = Flask(__name__)
CORS(app)

def encode_image(image_path):
    """Convert image to base64 format"""
    with open(image_path, "rb") as image_file:
        return base64.b64encode(image_file.read()).decode("utf-8")

def analyze_chart(image_path):
    """Analyze trading chart using Gemini API"""
    base64_image = encode_image(image_path)

    prompt_text = """
    You are an AI trained in financial markets. Analyze the uploaded chart and provide structured insights:

    1️⃣ Coin Name & Price  
    2️⃣ Strong Support & Resistance  
        - Weak support & resistance  
        - SMC Concept  
        - Break of structure & Change of character (ChoCH)  
        - Major levels for buy traps & sell traps  
    3️⃣ Market Condition (Bullish/Bearish/Sideways)  
    4️⃣ Future Trend Prediction  
    5️⃣ Entry & Exit Strategy  
    6️⃣ Stoploss & Target  
    7️⃣ Risk-Reward Ratio  
    8️⃣ EMA Analysis (9 EMA, 21 EMA, 50 EMA, 200 EMA)  
    9️⃣ RSI Condition (Overbought/Oversold/Divergence)  

    Provide insights in a structured JSON format.
    """

    model = genai.GenerativeModel("gemini-pro-vision")
    response = model.generate_content(
        [prompt_text, {"mime_type": "image/png", "data": base64_image}]
    )

    return response.text  # Extract response text from Gemini API

@app.route("/analyze", methods=["POST"])
def analyze():
    file = request.files["file"]
    file_path = "temp_chart.png"
    file.save(file_path)

    try:
        result = analyze_chart(file_path)
        return jsonify({"result": result})
    except Exception as e:
        return jsonify({"error": str(e)}), 500

if __name__ == "__main__":
    app.run(debug=True, port=5000)
