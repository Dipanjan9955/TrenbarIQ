// Update the showAnalysisModal function to better handle different data formats
function showAnalysisModal(analysisData, symbol, timeframe) {
  // Create modal element
  const modal = document.createElement('div');
  modal.className = 'modal-overlay';
  
  let analysisContent = '';
  
  try {
    // Format the analysis data
    if (typeof analysisData === 'string') {
      // Try to parse string as JSON
      try {
        const jsonData = JSON.parse(analysisData);
        analysisContent = formatJsonAnalysis(jsonData);
      } catch (e) {
        // Display as plain text if JSON parsing fails
        analysisContent = `<div class="analysis-section">
          <h3>Analysis Result</h3>
          <div class="analysis-details">
            <p>${analysisData}</p>
          </div>
        </div>`;
      }
    } else if (typeof analysisData === 'object' && analysisData !== null) {
      // Object data
      analysisContent = formatJsonAnalysis(analysisData);
    } else {
      // Fallback for unknown data types
      analysisContent = `<div class="analysis-section">
        <h3>Analysis Result</h3>
        <div class="analysis-details">
          <p>No details available</p>
        </div>
      </div>`;
    }
  } catch (error) {
    console.error("Error formatting analysis data:", error);
    analysisContent = `<div class="analysis-section">
      <h3>Error</h3>
      <div class="analysis-details">
        <p>An error occurred while formatting the analysis data.</p>
      </div>
    </div>`;
  }
  
  modal.innerHTML = `
    <div class="modal-container">
      <div class="modal-header">
        <h3>${symbol || 'Chart Analysis'} (${timeframe || 'N/A'})</h3>
        <button class="modal-close">&times;</button>
      </div>
      <div class="modal-content analysis-content">
        ${analysisContent}
      </div>
    </div>
  `;
  
  document.body.appendChild(modal);
  
  // Add close button event listener
  modal.querySelector('.modal-close').addEventListener('click', function() {
    modal.remove();
  });
  
  // Close on outside click
  modal.addEventListener('click', function(e) {
    if (e.target === modal) {
      modal.remove();
    }
  });
}

// Helper function to format JSON analysis data
function formatJsonAnalysis(data) {
  if (!data || typeof data !== 'object') {
    return '<p>No data available</p>';
  }
  
  return Object.entries(data).map(([key, value], index) => {
    let formattedValue = '';
    
    if (typeof value === 'object' && value !== null) {
      formattedValue = `<pre class="json-code">${JSON.stringify(value, null, 2)}</pre>`;
    } else if (typeof value === 'string') {
      // Check if the string contains line breaks
      if (value.includes('\n')) {
        formattedValue = `<p class="multiline-text">${value.replace(/\n/g, '<br>')}</p>`;
      } else {
        formattedValue = `<p>${value}</p>`;
      }
    } else {
      formattedValue = `<p>${value}</p>`;
    }
    
    return `
      <div class="analysis-section json-block" style="--index: ${index}">
        <h3>${key.replace(/_/g, ' ').replace(/\b\w/g, c => c.toUpperCase())}</h3>
        <div class="analysis-details">
          ${formattedValue}
        </div>
      </div>
    `;
  }).join('');
}
