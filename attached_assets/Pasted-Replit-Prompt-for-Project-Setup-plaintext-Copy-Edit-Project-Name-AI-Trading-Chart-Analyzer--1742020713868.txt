import google.generativeai as genai
import base64
from flask import Flask, request, jsonify
from flask_cors import CORS

# 🔹 Configure Gemini AI API Key
genai.configure(api_key="YOUR_GEMINI_API_KEY")

app = Flask(__name__)
CORS(app)

def encode_image(image_path):
    """Convert image to base64 for API"""
    with open(image_path, "rb") as image_file:
        return base64.b64encode(image_file.read()).decode("utf-8")

def analyze_chart(image_path):
    """Analyze trading chart using Gemini API"""
    base64_image = encode_image(image_path)

    prompt = """
    You are a trading AI expert. Analyze the uploaded chart and provide insights:

    **🔹 Market Analysis:**  
    1️⃣ Coin Name & Price  
    2️⃣ Strong & Weak Support/Resistance Levels  
    3️⃣ SMC Concepts (Break of Structure, Change of Character)  
    4️⃣ Key Liquidity Zones (Buy/Sell Traps)  
    5️⃣ Market Condition (Bullish/Bearish/Sideways)  
    6️⃣ Future Trend Prediction  
    7️⃣ Entry & Exit Strategy  
    8️⃣ Stoploss & Take Profit Targets  
    9️⃣ Risk-Reward Ratio  
    🔟 EMA Analysis (9, 21, 50, 200 EMA)  
    11️⃣ RSI Condition (Overbought/Oversold/Divergence)  
    12️⃣ Bollinger Bands & Volume Spikes  
    13️⃣ Candlestick Pattern Recognition  
    14️⃣ Institutional Accumulation/Distribution Analysis  

    **🔹 Return structured JSON output for better readability.**
    """

    response = genai.chat(model="gemini-pro-vision", messages=[
        {"role": "user", "content": [{"type": "text", "text": prompt}, 
                                     {"type": "image", "source": {"mime_type": "image/png", "data": base64_image}}]}
    ])

    return response.last.text  # Gemini output

@app.route("/analyze", methods=["POST"])
def analyze():
    file = request.files["file"]
    file_path = "temp_chart.png"
    file.save(file_path)

    result = analyze_chart(file_path)
    return jsonify({"result": result})

if __name__ == "__main__":
    app.run(debug=True, port=5000)

document.getElementById("upload").addEventListener("change", async (event) => {
    const file = event.target.files[0];
    if (!file) return;

    // Preview Image
    const preview = document.getElementById("preview");
    preview.src = URL.createObjectURL(file);
    preview.style.display = "block";

    // Upload Image to Backend
    let formData = new FormData();
    formData.append("file", file);

    const response = await fetch("https://REPLIT-PROJECT-URL/analyze", {
        method: "POST",
        body: formData,
    });

    const data = await response.json();
    document.getElementById("output").innerText = JSON.stringify(data.result, null, 2);
});
