// Profile tab functionality
// Wait for DOM content to load
document.addEventListener('DOMContentLoaded', function() {
  const profileTab = document.getElementById('profile-tab');
  if (!profileTab) return;

  // Check authentication status
  fetch('/user-status')
    .then(response => response.json())
    .then(data => {
      if (data.status === 'authenticated') {
        loadUserProfile(data.id);
      } else {
        showLoginPrompt();
      }
    })
    .catch(error => {
      console.error('Error loading profile:', error);
      profileTab.innerHTML = `
        <div class="card">
          <div class="card-body text-center">
            <h3>Error Loading Profile</h3>
            <p>Please try again later.</p>
          </div>
        </div>
      `;
    });
});

function loadUserProfile(userId) {

    fetch('/get-analyses').then(res => res.json()),

  ])
  .then(([preferences, analyses, backtests]) => {
    const profileTab = document.getElementById('profile-tab');

}

// Load user profile data
function loadUserProfile() {
  const profileContainer = document.getElementById('profile-content');
  if (!profileContainer) return;
  
  // Show loader
  profileContainer.innerHTML = `
    <div class="loader-container">
      <div class="loader active"></div>
      <div class="loader-text">Loading profile data...</div>
    </div>
  `;
  
  // Check user authentication status
  fetch('/user-status')
    .then(response => response.json())
    .then(data => {
      if (data.status === 'authenticated') {
        // User is authenticated, load profile data
        loadAuthenticatedProfile(data.username);
      } else if (data.status === 'guest') {
        // Guest user
        displayGuestProfile();
      } else {
        // Not authenticated
        displayLoginPrompt();
      }
    })
    .catch(error => {
      console.error('Error fetching user status:', error);
      profileContainer.innerHTML = `
        <div class="alert alert-danger">
          Error loading profile: ${error.message}
        </div>
      `;
    });
}

// Display data for authenticated user
function loadAuthenticatedProfile(username) {
  const profileContainer = document.getElementById('profile-content');
  
  // Load user preferences
  fetch('/get-preferences')
    .then(response => {
      if (!response.ok) {
        throw new Error('Failed to fetch preferences');
      }
      return response.json();
    })
    .then(preferences => {
      // Load user analysis history
      return fetch('/get-analyses')
        .then(response => {
          if (!response.ok) {
            throw new Error('Failed to fetch analyses');
          }
          return response.json();
        })
        .then(analyses => {
          // Load backtest results
          return fetch('/get-backtest-results')
            .then(response => {
              if (!response.ok) {
                throw new Error('Failed to fetch backtest results');
              }
              return response.json();
            })
            .then(backtestResults => {
              // Combine all data and display profile
              displayUserProfile(username, preferences, analyses, backtestResults);
            });
        });
    })
    .catch(error => {
      console.error('Error loading profile data:', error);
      profileContainer.innerHTML = `
        <div class="alert alert-danger">
          Error loading profile data: ${error.message}
        </div>
      `;
    });
}

// Display the full user profile
function displayUserProfile(username, preferences, analyses, backtestResults) {
  const profileContainer = document.getElementById('profile-content');
  
  // Calculate stats
  const totalAnalyses = analyses.length;
  const totalBacktests = backtestResults.length;
  const profitableBacktests = backtestResults.filter(result => result.profit_loss > 0).length;
  const winRate = totalBacktests > 0 ? 
    Math.round((profitableBacktests / totalBacktests) * 100) : 0;
  
  // Format date
  const formatDate = (dateString) => {
    const date = new Date(dateString);
    return date.toLocaleDateString() + ' ' + date.toLocaleTimeString();
  };
  
  // Build HTML
  let html = `
    <div class="profile-header">
      <div class="profile-avatar">
        ${username.charAt(0).toUpperCase()}
      </div>
      <div class="profile-info">
        <h2>${username}</h2>
        <p>Member since ${new Date().toLocaleDateString()}</p>
      </div>
    </div>
    
    <div class="profile-stats">
      <div class="stat-item">
        <div class="stat-value">${totalAnalyses}</div>
        <div class="stat-label">Chart Analyses</div>
      </div>
      <div class="stat-item">
        <div class="stat-value">${totalBacktests}</div>
        <div class="stat-label">Backtests</div>
      </div>
      <div class="stat-item">
        <div class="stat-value">${winRate}%</div>
        <div class="stat-label">Strategy Win Rate</div>
      </div>
    </div>
  `;
  
  // User preferences section
  html += `
    <div class="card mb-4">
      <div class="card-header">
        <h3><i class="fas fa-cog me-2"></i> Preferences</h3>
      </div>
      <div class="card-body">
        <form id="preferences-form">
          <div class="row">
            <div class="col-md-6">
              <div class="form-group">
                <label class="form-label">Theme</label>
                <select class="form-control" name="theme">
                  <option value="dark" ${preferences.theme === 'dark' ? 'selected' : ''}>Dark</option>
                  <option value="light" ${preferences.theme === 'light' ? 'selected' : ''}>Light</option>
                </select>
              </div>
            </div>
            <div class="col-md-6">
              <div class="form-group">
                <label class="form-label">Default Symbol</label>
                <input type="text" class="form-control" name="default_symbol" 
                  value="${preferences.default_symbol || 'BTCUSDT'}">
              </div>
            </div>
          </div>
          
          <div class="row mt-3">
            <div class="col-md-6">
              <div class="form-group">
                <label class="form-label">Default Timeframe</label>
                <select class="form-control" name="default_timeframe">
                  <option value="1m" ${preferences.default_timeframe === '1m' ? 'selected' : ''}>1 Minute</option>
                  <option value="5m" ${preferences.default_timeframe === '5m' ? 'selected' : ''}>5 Minutes</option>
                  <option value="15m" ${preferences.default_timeframe === '15m' ? 'selected' : ''}>15 Minutes</option>
                  <option value="1h" ${preferences.default_timeframe === '1h' ? 'selected' : ''}>1 Hour</option>
                  <option value="4h" ${preferences.default_timeframe === '4h' ? 'selected' : ''}>4 Hours</option>
                  <option value="1d" ${preferences.default_timeframe === '1d' ? 'selected' : ''}>1 Day</option>
                  <option value="1w" ${preferences.default_timeframe === '1w' ? 'selected' : ''}>1 Week</option>
                </select>
              </div>
            </div>
            <div class="col-md-6">
              <div class="form-group mt-4">
                <div class="form-check form-switch">
                  <input class="form-check-input" type="checkbox" name="notifications_enabled" 
                    ${preferences.notifications_enabled ? 'checked' : ''}>
                  <label class="form-check-label">Enable Notifications</label>
                </div>
              </div>
              <div class="form-group mt-2">
                <div class="form-check form-switch">
                  <input class="form-check-input" type="checkbox" name="ui_animations" 
                    ${preferences.ui_animations ? 'checked' : ''}>
                  <label class="form-check-label">Enable UI Animations</label>
                </div>
              </div>
            </div>
          </div>
          
          <button type="submit" class="btn btn-primary mt-4">
            <i class="fas fa-save me-2"></i> Save Preferences
          </button>
        </form>
      </div>
    </div>
  `;
  
  // Recent analyses section
  html += `
    <div class="card mb-4">
      <div class="card-header">
        <h3><i class="fas fa-chart-line me-2"></i> Recent Analyses</h3>
      </div>
      <div class="card-body">
  `;
  
  if (analyses.length > 0) {
    html += `<div class="table-responsive">
      <table class="table table-hover">
        <thead>
          <tr>
            <th>Date</th>
            <th>Symbol</th>
            <th>Timeframe</th>
            <th>Actions</th>
          </tr>
        </thead>
        <tbody>
    `;
    
    // Add up to 5 most recent analyses
    analyses.slice(0, 5).forEach(analysis => {
      html += `
        <tr>
          <td>${formatDate(analysis.created_at)}</td>
          <td>${analysis.symbol || 'Unknown'}</td>
          <td>${analysis.timeframe || '-'}</td>
          <td>
            <button class="btn btn-sm btn-primary view-analysis" data-id="${analysis.id}">
              <i class="fas fa-eye"></i> View
            </button>
          </td>
        </tr>
      `;
    });
    
    html += `
        </tbody>
      </table>
    </div>`;
    
    if (analyses.length > 5) {
      html += `<button class="btn btn-outline-primary mt-3">
        <i class="fas fa-list"></i> View All (${analyses.length})
      </button>`;
    }
  } else {
    html += `
      <div class="alert alert-info">
        <i class="fas fa-info-circle me-2"></i> You haven't saved any analyses yet.
      </div>
    `;
  }
  
  html += `
      </div>
    </div>
  `;
  
  // Backtest results section
  html += `
    <div class="card">
      <div class="card-header">
        <h3><i class="fas fa-flask me-2"></i> Backtest Results</h3>
      </div>
      <div class="card-body">
  `;
  
  if (backtestResults.length > 0) {
    html += `<div class="table-responsive">
      <table class="table table-hover">
        <thead>
          <tr>
            <th>Date</th>
            <th>Asset</th>
            <th>Strategy</th>
            <th>Profit/Loss</th>
            <th>Win Rate</th>
            <th>Actions</th>
          </tr>
        </thead>
        <tbody>
    `;
    
    // Add up to 5 most recent backtest results
    backtestResults.slice(0, 5).forEach(result => {
      const isProfitable = result.profit_loss > 0;
      const resultClass = isProfitable ? 'text-success' : 'text-danger';
      const resultSign = isProfitable ? '+' : '';
      
      html += `
        <tr>
          <td>${formatDate(result.created_at)}</td>
          <td>${result.asset}</td>
          <td>${result.strategy}</td>
          <td class="${resultClass}">
            ${resultSign}${result.profit_loss_percent.toFixed(2)}%
          </td>
          <td>${result.win_rate.toFixed(1)}%</td>
          <td>
            <button class="btn btn-sm btn-primary view-backtest" data-id="${result.id}">
              <i class="fas fa-eye"></i> View
            </button>
          </td>
        </tr>
      `;
    });
    
    html += `
        </tbody>
      </table>
    </div>`;
    
    if (backtestResults.length > 5) {
      html += `<button class="btn btn-outline-primary mt-3">
        <i class="fas fa-list"></i> View All (${backtestResults.length})
      </button>`;
    }
  } else {
    html += `
      <div class="alert alert-info">
        <i class="fas fa-info-circle me-2"></i> You haven't saved any backtest results yet.
      </div>
    `;
  }
  
  html += `
      </div>
    </div>
  `;
  
  // Update the container
  profileContainer.innerHTML = html;
  
  // Add event listeners for the preferences form
  const preferencesForm = document.getElementById('preferences-form');
  if (preferencesForm) {
    preferencesForm.addEventListener('submit', function(e) {
      e.preventDefault();
      
      // Collect form data
      const formData = new FormData(preferencesForm);
      const preferences = {
        theme: formData.get('theme'),
        default_symbol: formData.get('default_symbol'),
        default_timeframe: formData.get('default_timeframe'),
        notifications_enabled: formData.get('notifications_enabled') === 'on',
        ui_animations: formData.get('ui_animations') === 'on'
      };
      
      // Save preferences
      fetch('/save-preferences', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(preferences)
      })
        .then(response => response.json())
        .then(data => {
          if (data.success) {
            showFlashMessage('Preferences saved successfully', 'success');
          } else {
            showFlashMessage('Error saving preferences', 'error');
          }
        })
        .catch(error => {
          console.error('Error saving preferences:', error);
          showFlashMessage('Error saving preferences: ' + error.message, 'error');
        });
    });
  }
}
